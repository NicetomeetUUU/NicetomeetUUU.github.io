## 互斥锁的使用（C++11）
```
#include <iostream>
#include <thread>
#include <mutex>

std::mutex mtx; // 全局互斥锁.
int count = 0; // 全局共享资源.

void func() {
    for (int i = 0; i < 100000; ++i) {
        std::lock_guard<std::mutex> lock(mtx); // 自动获取互斥锁.
        ++count; // 操作共享资源.
        // 当lock对象析构时，自动释放互斥锁.
    }
}

int main() {
    std::thread t1(func);
    std::thread t2(func);

    t1.join();
    t2.join();

    std::cout << count << std::endl; // 输出200000

    return 0;
}
```
